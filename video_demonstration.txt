    def should_interrupt_agent(self, text: str, agent_is_speaking: bool) -> bool:
        # Check commands FIRST (highest priority)
        if self.contains_command(text):
            return True  # Always interrupt on commands
        
        # If no command, check if agent is speaking
        if agent_is_speaking:
            # If agent speaking, only interrupt if NOT backchanneling
            return not self.is_backchanneling(text)
        
        # Agent not speaking, allow normal interruption
        return True
